// Проблема: объект имеет несколько состояний 
// Задача: состояния игрока - покой, идти, бегать 

// Пример плохого кода: много "если" 
класс Игрок
 переменная состояниеИгрока // - покой, идти, бегать

 действие Обновить() //  вызывается раз в кадор 
  если состояниеИгрока == покой
   БытьВПокое()

  если состояниеИгрока == идти
   ИгрокИдет()

  если состояниеИгрока == бегать
   ИгрокБежим()



// Решение: паттерн состояние 
класс Состояние
 действие Вход()
 действие Выполнить()
 действие Выход()

класс СостояниеПокоя : Состояние 
 переопределить действие Вход()
 переопределить действие Выполнить()
 переопределить действие Выход()

класс СостояниеИдти : Состояние 
 переопределить действие Вход()
 переопределить действие Выполнить()
 переопределить действие Выход()

класс СостояниеБежать : Состояние 
 переопределить действие Вход()
 переопределить действие Выполнить()
 переопределить действие Выход()

класс Игрок 
 переменная Состояние текущееСостояние

 действие Обновить() //  вызывается раз в кадор 
  текущееСостояние.Выполнить()

 действие ИзменитьСостояние(Состояние новоеСостояние)
  текущееСостояние.Выход()
  текущееСостояние = новоеСостояние
  текущееСостояние.Вход()


// Пример использования 
Игрок игрок 

игрок.Обновить() // например вызывается каждый кадр 

игрок.ИзменитьСостояние(СостояниеИдти)
игрок.ИзменитьСостояние(СостояниеБежать)
игрок.ИзменитьСостояние(СостояниеПокоя)
