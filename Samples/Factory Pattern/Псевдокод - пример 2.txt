// Виды транспорта
класс Транспорт 
 действие Доставка()

класс Грузовик : Транспорт 
 перегрузка действие Доставка()

класс Корабль : Транспорт 
 перегрузка действие Доставка()

// Способы доставки (логистика) - фабричный метод 
класс Логистика 
 действие СоздатьТранспорт()
 действие ВыполнитьДоставку()

// Конкретные фабрики 
класс ЛогистикаПоСуши : Логистика
 перегрузка действие СоздатьТранспорт()
 перегрузка действие ВыполнитьДоставку()

класс ЛогистикаПоМорю : Логистика
 перегрузка действие СоздатьТранспорт()
 перегрузка действие ВыполнитьДоставку()


// Пример использования:
Логистка логистика_по_суши = новая ЛогистикаПоСуши()
логистика_по_суши.СоздатьТранспорт()
логистика_по_суши.ВыполнитьДоставку()

// Добавление новой логистики не требует изменение старого кода: 
Логистка логистика_по_морю = новая ЛогистикаПоМорю()
